Entrar:
npm.cmd run dev
$credenciales = @{
  nombreUsuario = "admin"
  password = "TU_CONTRASEÑA" 
} | ConvertTo-Json

$login = Invoke-RestMethod `
  -Method Post `
  -Uri "http://localhost:3000/api/admin/login" `
  -ContentType "application/json" `
  -Body $credenciales

$login.data.administrador
$token = $login.data.token



Reserva ocupada:
$confirmacion = @{
  fecha = "2026-07-25"
  hora = "11:00"
  nota = "Confirmado por WhatsApp"
} | ConvertTo-Json

Invoke-RestMethod `
  -Method Patch `
  -Uri "http://localhost:3000/api/admin/turnos/ocupar" `
  -Headers @{ Authorization = "Bearer $token" } `
  -ContentType "application/json" `
  -Body $confirmacion
